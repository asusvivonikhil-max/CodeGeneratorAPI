import java.util.Scanner;

public class Table {
    public static void main(String[] args) {
        int n=0;
        Scanner scanner = new Scanner(System.in);
        
        System.out.print("Enter a Number = ");
        n=scanner.nextInt();
        for(int i=1;i<=10;i++){
            System.out.println(n+" * "+i+" = "+(n*i));
        }
        
        scanner.close();
    }
}
